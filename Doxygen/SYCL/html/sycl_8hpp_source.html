814 template <int Dimensions = 1, typename ParallelForFunctor>
815 void parallel_for(nd_range<Dimensions> r,
816  ParallelForFunctor f) {
817  // In a sequential execution there is only one index processed at a time
818  item<Dimensions> Index { r };
819  // To iterate on the work-group
820  id<Dimensions> Group;
821  range<Dimensions> GroupRange = r.get_group_range();
822  // To iterate on the local work-item
823  id<Dimensions> Local;
824  range<Dimensions> LocalRange = r.get_local_range();
825 
826  // Reconstruct the item from its group and local id
827  auto reconstructItem = [&] (id<Dimensions> L) {
828  //Local.display();
829  // Reconstruct the global item
830  Index.set_local(Local);
831  Index.set_global(Local + LocalRange*Group);
832  // Call the user kernel at last
833  f(Index);
834  };

837  into another functor that iterate inside the work-group and then
838  calls f */
839  auto iterateInWorkGroup = [&] (id<Dimensions> G) {
840  //Group.display();
841  // Then iterate on the local work-groups

844  decltype(reconstructItem),
