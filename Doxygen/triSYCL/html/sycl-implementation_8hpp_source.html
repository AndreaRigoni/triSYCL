451 /** A recursive multi-dimensional iterator that ends calling f
452 
453  The iteration order may be changed later.
454 
455  Since partial specialization of function template is not possible in
456  C++14, use a class template instead with everything in the
457  constructor.
458 */
459 template <int level, typename Range, typename ParallelForFunctor, typename Id>
460 struct ParallelForIterate {
461  ParallelForIterate(const Range &r, ParallelForFunctor &f, Id &index) {
462  for (boost::multi_array_types::index _sycl_index = 0,
463  _sycl_end = r[Range::dimensionality - level];
464  _sycl_index < _sycl_end;
465  _sycl_index++) {
466  // Set the current value of the index for this dimension
467  index[Range::dimensionality - level] = _sycl_index;
468  // Iterate further on lower dimensions
469  ParallelForIterate<level - 1,
470  Range,
471  ParallelForFunctor,
472  Id> { r, f, index };
473  }
474  }
475 };

478 /** A top-level recursive multi-dimensional iterator variant using OpenMP
479 
480  Only the top-level loop uses OpenMP and go on with the normal
481  recursive multi-dimensional.
482 */
483 template <int level, typename Range, typename ParallelForFunctor, typename Id>
484 struct ParallelOpenMPForIterate {
485  ParallelOpenMPForIterate(const Range &r, ParallelForFunctor &f) {
486  // Create the OpenMP threads before the for loop to avoid creating an
487  // index in each iteration
488 #pragma omp parallel
489  {
490  // Allocate an OpenMP thread-local index
491  Id index;
492  // Make a simple loop end condition for OpenMP
493  boost::multi_array_types::index _sycl_end =
494  r[Range::dimensionality - level];
495  /* Distribute the iterations on the OpenMP threads. Some OpenMP
496  "collapse" could be useful for small iteration space, but it
497  would need some template specialization to have real contiguous
498  loop nests */
499 #pragma omp for
500  for (boost::multi_array_types::index _sycl_index = 0;
501  _sycl_index < _sycl_end;
502  _sycl_index++) {
503  // Set the current value of the index for this dimension
504  index[Range::dimensionality - level] = _sycl_index;
505  // Iterate further on lower dimensions
506  ParallelForIterate<level - 1,
507  Range,
508  ParallelForFunctor,
509  Id> { r, f, index };
510  }
511  }
512  }
513 };
